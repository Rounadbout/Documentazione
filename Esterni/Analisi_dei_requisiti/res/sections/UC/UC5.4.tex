\subsubsection{UC5.4 - Visualizzazione errore: formato private key errato}
\begin{itemize}
	\item \textbf{Attori primari:} \una{};
	\item \textbf{Descrizione:} l’utente tenta di eseguire la procedura di login con una private key\ped{\textit{G}} non conforme al formato richiesto da Ethereum\ped{\textit{G}}; 
	\item \textbf{Scenario principale:} il sistema mostra un messaggio di errore relativo al formato della private key\ped{\textit{G}} inserita;
	\item \textbf{Precondizione:} l'utente ha inserito una private key\ped{\textit{G}} in formato errato durante la procedura di autenticazione;
	\item \textbf{Postcondizione:} viene mostrato all’utente un messaggio di errore. 
\end{itemize}
